#include // LCD LiquidCrystal lcd(8, 9, 10, 11, 12, 13); // Ultrassônico #define TRIG1 22 #define ECHO1 23 #define TRIG2 24 #define ECHO2 25 // Ponte H #define IN1 6 #define IN2 7 #define ENA 5 // Parâmetros dos tanques (em cm) float alturaTanque = 30.0; // Histerese float nivelMinTanque1 = 5.0; // mínimo seguro float nivelRetorno = 10.0; // nível para voltar a ligar float nivelLigaTanque2 = 20.0; float nivelDesligaTanque2 = 5.0; bool bloqueioSeguranca = false; long lerUltrassom(int trig, int echo) { digitalWrite(trig, LOW); delayMicroseconds(2); digitalWrite(trig, HIGH); delayMicroseconds(10); digitalWrite(trig, LOW); long duracao = pulseIn(echo, HIGH); long distancia = duracao * 0.034 / 2; return distancia; } float calcularNivel(long distancia) { float nivel = alturaTanque - distancia; if (nivel < 0) nivel = 0; if (nivel > alturaTanque) nivel = alturaTanque; return nivel; } void setup() { pinMode(TRIG1, OUTPUT); pinMode(ECHO1, INPUT); pinMode(TRIG2, OUTPUT); pinMode(ECHO2, INPUT); pinMode(IN1, OUTPUT); pinMode(IN2, OUTPUT); pinMode(ENA, OUTPUT); lcd.begin(16, 2); lcd.print("Sistema Agua"); delay(2000); lcd.clear(); } void loop() { long dist1 = lerUltrassom(TRIG1, ECHO1); long dist2 = lerUltrassom(TRIG2, ECHO2); float nivel1 = calcularNivel(dist1); float nivel2 = calcularNivel(dist2); // --- SEGURANÇA (Tanque 1) --- if (nivel1 < nivelMinTanque1) { bloqueioSeguranca = true; } if (nivel1 > nivelRetorno) { bloqueioSeguranca = false; } // --- CONTROLE DA BOMBA --- int velocidade = 0; if (!bloqueioSeguranca) { if (nivel2 < nivelLigaTanque2) { // velocidade proporcional velocidade = map(nivel2, 0, alturaTanque, 255, 100); } if (nivel2 <= nivelDesligaTanque2) { velocidade = 0; } } else { velocidade = 0; } // Aciona ponte H if (velocidade > 0) { digitalWrite(IN1, HIGH); digitalWrite(IN2, LOW); analogWrite(ENA, velocidade); } else { digitalWrite(IN1, LOW); digitalWrite(IN2, LOW); analogWrite(ENA, 0); } // --- LCD --- lcd.setCursor(0, 0); lcd.print("T1:"); lcd.print(nivel1); lcd.print("cm "); lcd.setCursor(0, 1); lcd.print("T2:"); lcd.print(nivel2); lcd.print("cm "); if (bloqueioSeguranca) { lcd.setCursor(10, 1); lcd.print("ALERTA"); } else if (velocidade > 0) { lcd.setCursor(10, 1); lcd.print("BOMBA"); } else { lcd.setCursor(10, 1); lcd.print("PARADA"); } delay(500); }
