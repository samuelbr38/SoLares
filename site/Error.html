#include #define TRIG1 22 #define ECHO1 23 #define TRIG2 24 #define ECHO2 25 #define ENA 5 #define IN1 6 #define IN2 7 LiquidCrystal_I2C lcd(0x27, 16, 2); float nivel1 = 0; float nivel2 = 0; float alturaMax = 30.0; float ligaBomba = 40.0; float desligaBomba = 80.0; float minimoSeguro = 20.0; long readUltrasonic(int trig, int echo) { digitalWrite(trig, LOW); delayMicroseconds(2); digitalWrite(trig, HIGH); delayMicroseconds(10); digitalWrite(trig, LOW); long duration = pulseIn(echo, HIGH); return duration * 0.034 / 2; } float calculaNivel(long distancia) { float nivel = ((alturaMax - distancia) / alturaMax) * 100.0; if (nivel > 100) nivel = 100; if (nivel < 0) nivel = 0; return nivel; } void setup() { pinMode(TRIG1, OUTPUT); pinMode(ECHO1, INPUT); pinMode(TRIG2, OUTPUT); pinMode(ECHO2, INPUT); pinMode(ENA, OUTPUT); pinMode(IN1, OUTPUT); pinMode(IN2, OUTPUT); lcd.init(); lcd.backlight(); digitalWrite(IN1, HIGH); digitalWrite(IN2, LOW); } void loop() { long dist1 = readUltrasonic(TRIG1, ECHO1); long dist2 = readUltrasonic(TRIG2, ECHO2); nivel1 = calculaNivel(dist1); nivel2 = calculaNivel(dist2); int velocidade = map(nivel1, 0, 100, 255, 80); bool bombaLigada = false; if (nivel2 < ligaBomba) { bombaLigada = true; } if (nivel2 > desligaBomba) { bombaLigada = false; } if (nivel1 < minimoSeguro) { bombaLigada = false; analogWrite(ENA, 0); lcd.clear(); lcd.setCursor(0, 0); lcd.print("FALTA AGUA T1"); lcd.setCursor(0, 1); lcd.print("BOMBA OFF"); delay(1000); return; } if (bombaLigada) { analogWrite(ENA, velocidade); } else { analogWrite(ENA, 0); } lcd.clear(); lcd.setCursor(0, 0); lcd.print("T1:"); lcd.print(nivel1); lcd.print("%"); lcd.setCursor(0, 1); lcd.print("T2:"); lcd.print(nivel2); lcd.print("%"); delay(1000); }
